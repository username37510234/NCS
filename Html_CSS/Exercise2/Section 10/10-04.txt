body {
         background-image: url(art.jpg);
         /* 배경 이미지를 적용하는 기본 방식 */
         background-size: 100% 500px;
         /* 배경 이미지 크기 100%는 브라우저의 가로 길이를 기준으로 100% */
         /* 100%(가로 크기), 500px(세로 크기) */
         background-repeat: no-repeat;
         /* 배경 이미지의 초기 반복 값은 가로, 세로 모두 반복
             no-repeat은 반복하지 않음
             repeat-x는 가로 반복
             repeat-y는 세로 반복 */
        }
</style>
